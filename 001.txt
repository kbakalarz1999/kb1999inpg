float funkcja (float a,float b,float c,float x)
{float y ;
 y=a*x*x+b*x+c ;
 return y;
         }

 float calka (float xp,float xk,int n, float a,float b,float c)
 {float y1,y2,dx,suma,x; int i;
 suma=0;
 dx=(xk-xp)/n;
 for (i=0;i<n;i++)
 {x=xp+i*dx;
 y1=funkcja(a,b,c,x);
 y2=funkcja(a,b,c,x+dx);
 suma=suma+((y1+y2)/2)*dx;
 }
 return suma;
 }
